const SPREADSHEET_ID = "17GQW-mUQPpk5Y0hrUTJE3yQTx9UJA8ditdOOd8lqrnU";
const SHEET_NAME = "REWORK_DATABASE";
const DEFECT_SHEET = "DEFECT_CODES";
const PHOTO_FOLDER_ID = "1FK5LMO7Yo6MEKcoAhqkjxA3a39pZXc1T";

function doGet(e) {

  if (e && e.parameter.action === "getDefects") {

    return ContentService
      .createTextOutput(
        JSON.stringify(getDefects())
      )
      .setMimeType(
        ContentService.MimeType.JSON
      );
  }

  return HtmlService
    .createHtmlOutput("BMW API OK");
}

function doPost(e) {

  const data = JSON.parse(
    e.postData.contents
  );

  const result = saveRecord(data);

  return ContentService
    .createTextOutput(
      JSON.stringify(result)
    )
    .setMimeType(
      ContentService.MimeType.JSON
    );
}

function getDefects() {

  const sheet =
    SpreadsheetApp
      .openById(
        SPREADSHEET_ID
      )
      .getSheetByName(
        DEFECT_SHEET
      );

  return sheet.getRange(
    2,
    1,
    sheet.getLastRow() - 1,
    2
  ).getValues();
}

function saveRecord(data) {

  const sheet =
    SpreadsheetApp
      .openById(
        SPREADSHEET_ID
      )
      .getSheetByName(
        SHEET_NAME
      );

  // DUPLICATE CHECK BY SERIAL NUMBER ONLY

  const rows =
    sheet.getDataRange()
      .getValues();

  for (let i = 1; i < rows.length; i++) {

    if (
      String(rows[i][2]).trim() ===
      String(data.serial).trim()
    ) {

      return {
        status: "DUPLICATE",
        id: "SERIAL_EXISTS"
      };

    }

  }

  let labelPhotoUrl = "";
  let photoUrl = "";
  let fullLabelText = "";

  // CUSTOMER LABEL PHOTO

  if (
    data.labelImage &&
    data.labelImage !== ""
  ) {

    const base64Data =
      data.labelImage.split(",")[1];

    const bytes =
      Utilities.base64Decode(
        base64Data
      );

    const fileName =
      "LABEL_" +
      data.serial +
      "_" +
      Utilities.formatDate(
        new Date(),
        Session.getScriptTimeZone(),
        "yyyyMMdd_HHmmss"
      ) +
      ".jpg";

    const blob =
      Utilities.newBlob(
        bytes,
        "image/jpeg",
        fileName
      );

    const folder =
      DriveApp.getFolderById(
        PHOTO_FOLDER_ID
      );

    const file =
      folder.createFile(blob);

    file.setSharing(
      DriveApp.Access.ANYONE_WITH_LINK,
      DriveApp.Permission.VIEW
    );

    labelPhotoUrl =
      file.getUrl();

    fullLabelText =
      extractLabelText(
        file.getId()
      );

  }

  // FAILURE PHOTO

  if (
    data.image &&
    data.image !== ""
  ) {

    const base64Data =
      data.image.split(",")[1];

    const bytes =
      Utilities.base64Decode(
        base64Data
      );

    const fileName =
      "FAILURE_" +
      data.serial +
      "_" +
      Utilities.formatDate(
        new Date(),
        Session.getScriptTimeZone(),
        "yyyyMMdd_HHmmss"
      ) +
      ".jpg";

    const blob =
      Utilities.newBlob(
        bytes,
        "image/jpeg",
        fileName
      );

    const folder =
      DriveApp.getFolderById(
        PHOTO_FOLDER_ID
      );

    const file =
      folder.createFile(blob);

    file.setSharing(
      DriveApp.Access.ANYONE_WITH_LINK,
      DriveApp.Permission.VIEW
    );

    photoUrl =
      file.getUrl();
  }

  const id =
    "RW-" +
    Utilities.formatString(
      "%06d",
      Math.max(
        sheet.getLastRow(),
        1
      )
    );

  sheet.appendRow([

    id,                 // A ID
    new Date(),         // B Date
    data.serial,        // C Serial Number
    data.defectCode,    // D Defect Code
    data.defectName,    // E Defect Description
    data.comment,       // F Comment
    data.qty,           // G Qty
    data.oem,           // H OEM
    labelPhotoUrl,      // I Customer Label
    photoUrl,           // J Failure Photo
    data.user,          // K User
    "OPEN",             // L Status
    fullLabelText       // M Full Label Text

  ]);

  return {

    status: "OK",
    id: id

  };

}

function extractLabelText(fileId) {

  try {

    const resource = {

      title:
        "OCR_" +
        new Date().getTime(),

      mimeType:
        MimeType.GOOGLE_DOCS

    };

    const ocrFile =
      Drive.Files.copy(
        resource,
        fileId,
        {
          ocr: true,
          ocrLanguage: "en"
        }
      );

    const text =
      DocumentApp
        .openById(
          ocrFile.id
        )
        .getBody()
        .getText();

    DriveApp
      .getFileById(
        ocrFile.id
      )
      .setTrashed(true);

    return text
      .replace(/\n/g, " | ")
      .replace(/\s+/g, " ")
      .trim();

  }
  catch(err){

    Logger.log(err);

    return "";

  }

}
